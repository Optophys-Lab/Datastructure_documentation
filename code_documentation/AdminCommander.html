"""
GUI to work with adminstrative stuff and lab schema tables
connects to DataJoint DB
Author: Artur

import json
import shutil

from PyQt6.QtWidgets import QApplication, QMainWindow, QFileDialog,QProgressDialog
from PyQt6.QtCore import Qt, QThreadPool

from PyQt6 import uic
import os, sys
from pymysql.err import OperationalError
from pathlib import Path
from datastructure_tools.DataBaseAccess import DataBaseAccess
from datastructure_tools.GUI_utils import ErrorDialog, Worker, InfoDialog
from datastructure_tools.eLabFTW_api_utils import HOST
from datastructure_tools.eLabFTW_API import eLabbookCommander

VERSION = "0.4.0"


class AdminCommander(QMainWindow):
    def __init__(self):
        super(AdminCommander, self).__init__()
        self.path2file = Path(os.path.abspath(__file__))
        uic.loadUi(os.path.join(self.path2file.parent,'ui', 'AdminCommander_design.ui'), self)
        self.setWindowTitle('AdminCommander v.%s' % VERSION)
        self.connectSignalsSlots()

        self.DB = DataBaseAccess()
        if not self.DB.server_settings_correct:
            for t_i in [1, 2, 3, 4, 5, 6]:
                self.Tabs.setTabEnabled(t_i, False)
            self.Tabs.setCurrentIndex(0)
        if not self.DB.is_connected:
            ErrorDialog('No connection to server possible')
            self.was_notconnected_init = True
        else:
            self.was_notconnected_init = False
            self.ReadDB()
            self.DBIP_edit_2.setText(self.DB.cfg['SQLserver_cfg']['host'])
            self.DBuser_edit_2.setText(self.DB.cfg['SQLserver_cfg']['user'])
            self.DBname_edit_2.setText(self.DB.cfg['SQLserver_cfg']['DataBase'])
        if not self.DB.server_path:
            ErrorDialog("No correct serverpath was found, Please Enter one and restart")
            self.ServerB.setEnabled(True)
            self.textBrowser.setEnabled(True)
            self.label_15.setEnabled(True)



        self.threadpool = QThreadPool()
        #print("Multithreading with maximum %d threads" % self.threadpool.maxThreadCount())
    def connectSignalsSlots(self):
        self.actionExit.triggered.connect(self.close)
        self.actionUpdateDB.triggered.connect(self.ReadDB)
        self.CUserB.clicked.connect(self.addUser2DB)
        self.ConfigB.clicked.connect(self.saveConfigs)
        self.ProjectB.clicked.connect(self.addProject2DB)
        self.BBlockB.clicked.connect(self.addBB2DB)
        self.ExpTypeB.clicked.connect(self.addExptype2DB)
        self.BBlockB_2.clicked.connect(self.addBBlock2widget)
        self.PathB.clicked.connect(self.browseBCallback)
        self.DBSaveB.clicked.connect(self.saveServerSettings)
        self.ServerB.clicked.connect(self.browseServerCallback)
        self.Project_combo_settings.currentIndexChanged.connect(self.changeExp)
        self.ExpType_combo.currentIndexChanged.connect(self.changeType)
        self.CopyAtlasesB.clicked.connect(self.copy_executer)
        # self.settingsD.buttonBox.accepted.connect(self.getInfosSettings)
        # self.projectD.buttonBox.accepted.connect(self.getInfosProject)
        # self.bblockD.buttonBox.accepted.connect(self.getInfosBBblock)
        self.APIKEY_saveButton.clicked.connect(self.saveAPIKEY)

    def populateGUI(self):
        self.ExistingUser_list.clear()
        self.ExistingUser_list.addItems(self.all_users)
        self.ExistingProject_list.clear()
        self.ExistingProject_list.addItems(self.all_projects)
        self.Existing_listBB.clear()
        self.Existing_listBB.addItems([item["building_block"] for item in self.all_bblocks])
        self.Existing_listBB_2.clear()
        self.Existing_listBB_2.addItems([item["raw"] for item in self.all_bblocks])
        self.Existing_listExp.clear()
        self.Existing_listExp.addItems([item["experiment_template"] for item in self.all_exptypes])
        self.Existing_listExp_2.clear()
        self.Existing_listExp_2.addItems([str(self.exp_templates[exp_type]) for exp_type in [item["experiment_template"]
                                                                                        for item in self.all_exptypes]])


        self.User_combo.clear()
        self.User_combo.addItems(self.all_users)
        self.Project_combo_settings.clear()
        self.Project_combo_settings.addItems(self.all_projects)
        self.ExpType_combo.clear()
        self.ExpType_combo.addItems([item["experiment_template"] for item in self.all_exptypes])
        self.BBlock_combo.clear()
        self.BBlock_combo.addItems([item["building_block"] for item in self.all_bblocks])
        if self.DB.server_path:
            potential_proj = [path for path in os.listdir(self.DB.server_path) if
                              ((Path(self.DB.server_path) / path).is_dir()) and (path not in self.all_projects)]
        else:
            potential_proj = None
        if not potential_proj:
            self.ProjectB.setDisabled(True)
            self.Project_combo.setDisabled(True)
        else:
            self.ProjectB.setDisabled(False)
            self.Project_combo.setDisabled(False)
            self.Project_combo.clear()
            self.Project_combo.addItems(potential_proj)

        self.Atlas_combo.clear()
        self.Atlas_combo.addItems([item["brain_atlas"] for item in self.all_atlases])

        self.eLabFTW_linkEdit.setText(f"{HOST.split('/api')[0]}/ucp.php?tab=4")
        if self.DB.cfg.get('eLabAPIKEY', None):
            self.elabAPIKEY.setText(self.DB.cfg.get('eLabAPIKEY'))

    def saveAPIKEY(self):
        APIKEY = self.elabAPIKEY.text()
        if eLabbookCommander(APIKEY, HOST,self.DB).test_connection():
            self.DB.cfg['eLabAPIKEY'] = APIKEY
            with open(self.path2file.parent / "user_config.json", "rt") as fi:
                user_config = json.load(fi)
            user_config.update({'eLabAPIKEY': APIKEY})
            with open(self.path2file.parent / "user_config.json", "wt") as fi:
                json.dump(user_config, fi)
            self.APIKEY_saveButton.setText('saved')
        else:
            ErrorDialog('Wrong APIKEY, check your input')


    def copy_complete(self):
        self.statusbar.showMessage('Finished coping atlas files')
        self.process_dialog.reset()
    def handle_copy_errors(self,e):
        if issubclass(e[0], FileExistsError):
            ErrorDialog('Atlas is already copied!')
        elif issubclass(e[0], FileNotFoundError):
            ErrorDialog('AtlasFiles not found!/nIs serverpath correct?')
    def copy_executer(self):
        self.process_dialog = QProgressDialog('Copying Atlas files..','Abort',0,0,autoClose=True)
        worker = Worker(self.copyAtlasFiles)  # Any other args, kwargs are passed to the run function
        worker.signals.error.connect(self.handle_copy_errors)
        worker.signals.finished.connect(self.copy_complete)
        self.threadpool.start(worker)
        self.statusbar.showMessage('Copying Atlas files..')
        self.process_dialog.exec()

    def copyAtlasFiles(self, progress_callback):
        if self.DB.server_settings_correct:
            atlas2copy = self.Atlas_combo.currentText()
            ATLAS_PATH = Path([item["atlas_path"] for item in self.all_atlases if item["brain_atlas"] == atlas2copy][0])
            if (Path(self.DB.server_path).parent.parent / ATLAS_PATH).exists():
                atlas_path = Path(self.DB.server_path).parent.parent / ATLAS_PATH
            else:
                atlas_path = Path(self.DB.server_path).parent / ATLAS_PATH
            # TODO remove single parent after changing the root directory

            cur_path = os.path.abspath(os.path.dirname(__file__))
            if not (Path(cur_path) / "Atlas_files").exists():
                os.mkdir(Path(cur_path) / "Atlas_files")

            QApplication.processEvents()
            shutil.copytree(atlas_path, Path(cur_path) / "Atlas_files" / atlas2copy)


    def changeType(self):
        try:
            bblock_available = self.exp_templates[self.ExpType_combo.currentText()]
            self.ExpType_combo_2.clear()
            self.ExpType_combo_2.addItems(bblock_available)
        except KeyError:  # stupid error due to change callbacks
            pass

    def changeExp(self):
        if self.DB.server_path:
            experiments_available = [path for path in
                                     os.listdir(self.DB.server_path / self.Project_combo_settings.currentText())
                                     if
                                     (self.DB.server_path / self.Project_combo_settings.currentText() / path).is_dir()]
        else:
            experiments_available = []
        for id, exp in enumerate(experiments_available):
            if exp == "misc":
                experiments_available.pop(id)
        self.ExpName_combo.clear()
        self.ExpName_combo.addItems(experiments_available)

    def addBBlock2widget(self):
        bbloc2add = self.BBlock_combo.currentText()
        if not self.BBblock_list.findItems(bbloc2add, Qt.MatchFlag.MatchExactly):
            self.BBblock_list.addItem(bbloc2add)

    def saveConfigs(self):
        if not self.DB.server_path:  # if not serverpath under defaults
            server_paths = self.DB.default_server_paths
            if not self.textBrowser.toPlainText():
                ErrorDialog("No serverpath added!")
                return
            else:
                server_paths.append(self.textBrowser.toPlainText())
        else:
            server_paths = self.DB.default_server_paths

        initialdir = {}
        for p in range(self.listWidget.count()):
            initialdir[self.listWidget.item(p).text()] = self.listWidget_2.item(p).text()
        user_config = {"Server_path": server_paths,
                       "user_id": self.User_combo.currentText(),
                       "project": self.Project_combo_settings.currentText(),
                       "experiment_template": self.ExpType_combo.currentText(),
                       "sub_experiment": self.ExpName_combo.currentText(),
                       "initialDir": initialdir}

        with open(self.path2file.parent / "user_config.json", "wt") as fi:
            json.dump(user_config, fi)

        self.statusbar.showMessage('User config file was saved')

    def saveServerSettings(self):
        if not self.DBIP_edit.text():
            ErrorDialog("No IP")
            return
        if not self.DBuser_edit.text():
            ErrorDialog("No user")
            return
        if not self.DBname_edit.text():
            ErrorDialog("No DB name !")
            return

        server_settings = {"host": self.DBIP_edit.text(), "user": self.DBuser_edit.text(),
                           "password": self.DBpass_edit.text(), "DataBase": self.DBname_edit.text()}
        settings = {"SQLserver_cfg": server_settings}

        import datajoint as dj
        dj.config['database.host'] = settings['SQLserver_cfg']['host']
        dj.config['database.user'] = settings['SQLserver_cfg']['user']
        dj.config['database.password'] = settings['SQLserver_cfg']['password']
        try:
            dj.conn(reset=True)
            self.DBIP_edit_2.setText(dj.config['database.host'])
            self.DBuser_edit_2.setText(dj.config['database.user'])
            self.DBname_edit_2.setText(settings['SQLserver_cfg']["DataBase"])
        except OperationalError:
            ErrorDialog('Wrong user or password? No connection possible')
            self.DBIP_edit_2.setText('Not Connected')
            self.DBuser_edit_2.setText('Not Connected')
            self.DBname_edit_2.setText('Not Connected')
            return
        with open(self.path2file.parent / 'DataJoint' / "server_config.json", "wt") as fi:
            json.dump(settings, fi)
        self.labelConnected.setText("Connected")
        self.statusbar.showMessage('DB config file was saved')

        if not self.was_notconnected_init:
            InfoDialog('server settings changed, restart application')
            # need to restart python, other all previous references to schemas are still pointing to old objects
            self.close()
        for t_i in [1, 2, 3, 4, 5, 6]:
            self.Tabs.setTabEnabled(t_i, True)

        self.DB.readConfigs()
        self.DB._connect()
        if self.DB.is_connected:  # only if connected proceed
            self.DB._instanciateSchemas()
        self.ReadDB()

    def browseServerCallback(self):
        path = QFileDialog.getExistingDirectory(self)
        self.textBrowser.setText(path)

    def browseBCallback(self):
        path = QFileDialog.getExistingDirectory(self)
        bbloc2add = self.ExpType_combo_2.currentText()
        if not self.listWidget.findItems(bbloc2add, Qt.MatchFlag.MatchExactly):
            self.listWidget.insertItem(0, bbloc2add)
            self.listWidget_2.insertItem(0, path)

    def addProject2DB(self):
        """
        is called when add Exp button is clicked
        checks inputs and sends them to DataBaseAccess
        """
        proj_name = self.Project_combo.currentText()
        proj_desc = self.ProjDesc_edit.toPlainText()
        keywords = [keyword.strip().lower() for keyword in self.ProjKey_edit.toPlainText().replace(',', '/').split('/')]

        if not proj_desc:
            ErrorDialog("No Description!")
            return

        if self.CreateFolderB.isChecked():
            folder_name = self.expName_Edit.text()
            if not folder_name:
                ErrorDialog("No folder name was given")
                return
            try:
                os.mkdir(self.DB.server_path / proj_name / folder_name)
            except FileExistsError:
                ErrorDialog("This folder exists already")
            except FileNotFoundError:
                ErrorDialog("Folder could not be created\nDo you have write rights ?\nCreate folder manually")

        if not self.DB.AddProject(proj_name, proj_desc, keywords):
            ErrorDialog('Something went wrong\n check terminal output')
        else:
            self.ReadDB()
            # self.populateGUI()
            self.ProjDesc_edit.clear()
            self.ProjKey_edit.clear()
            self.expName_Edit.clear()

    def addBB2DB(self):
        """
        is called when AddBB button is called
        checks inputs and sends them to DataBaseAccess
        """
        bb_name = self.BBname_edit.text()
        bb_desc = self.BBdesc_Edit.toPlainText()
        folder_edit = self.folder_edit.text()
        if self.folderMod_check.isChecked():
            folderPre_edit = self.folderPre_edit.text()
            folderProc_edit = self.folderProc_edit.text()
            if not folderProc_edit:
                folderProc_edit = None
        else:
            folderPre_edit = folder_edit
            folderProc_edit = None

        if not bb_name:
            ErrorDialog("No name Added")
            return
        if not bb_desc:
            ErrorDialog("No Description!")
            return
        if not folder_edit:
            ErrorDialog("No Raw Folder!")
            return

        if not folderPre_edit:
            ErrorDialog("No Preprocess Folder!")
            return

        if not self.DB.AddBBlock(bb_name, bb_desc, folder_edit, folderPre_edit, processed=folderProc_edit):
            ErrorDialog('Something went wrong\n check terminal output')
        else:
            self.ReadDB()
            # self.populateGUI()
            self.BBname_edit.clear()
            self.BBdesc_Edit.clear()
            self.folder_edit.clear()
            self.folderPre_edit.clear()
            self.folderProc_edit.clear()

    def addExptype2DB(self):
        """
        is called when add Exp buttun is clicked
        checks inputs and sends them to DataBaseAccess
        """
        exp_name = self.ExpType_edit.text()
        exp_desc = self.ExpDesc_edit.toPlainText()
        blocks = [self.BBblock_list.item(p).text() for p in range(self.BBblock_list.count())]
        if not exp_name:
            ErrorDialog("No name Added")
            return
        if not exp_desc:
            ErrorDialog("No Description!")
            return
        if not blocks:
            ErrorDialog("No Blocks Added!")
            return

        if not self.DB.AddExpType(exp_name, exp_desc, blocks):
            ErrorDialog('Something went wrong\n check terminal output')
        else:
            self.ReadDB()
            # self.populateGUI()
            self.BBblock_list.clear()
            self.ExpType_edit.clear()
            self.ExpDesc_edit.clear()

    def addUser2DB(self):
        """
          is called when add user button is clicked
          checks inputs and sends them to DataBaseAccess
          """
        if not self.userEdit.text():
            ErrorDialog('NO user!')
            return

        if not self.DB.AddUser(self.userEdit.text()):
            ErrorDialog('Something went wrong\n check terminal output')
        else:
            self.statusbar.showMessage("User %s added to DB" % self.userEdit.text())
            self.ReadDB()
            # self.populateGUI()
            self.userEdit.clear()

    def ReadDB(self):
        """
        accesses the DataJoint DB
        """
        self.all_projects = self.DB.Project.fetch("project")
        self.all_users = self.DB.User.fetch("user")
        self.all_exptypes = self.DB.ExperimentTemplate.fetch(as_dict=True)
        self.all_bblocks = self.DB.BuildingBlock.fetch(as_dict=True)
        self.all_atlases = self.DB.BrainAtlas.fetch(as_dict=True)
        self.exp_templates = {}
        for exp_type in self.DB.ExperimentTemplate.fetch('experiment_template'):
            self.exp_templates[exp_type] = list(
                (self.DB.ExperimentTemplate.TemplateBuild() & "experiment_template = '%s'" % '%s' % exp_type).fetch(
                    "building_block"))

        self.populateGUI()

def start_gui():
    app = QApplication([])
    win = AdminCommander()
    win.show()
    app.exec()

if __name__ == '__main__':
    sys.exit(start_gui())

def copy_executer(self):
    self.process_dialog = QProgressDialog('Copying Atlas files..','Abort',0,0,autoClose=True)
    worker = Worker(self.copyAtlasFiles)  # Any other args, kwargs are passed to the run function
    worker.signals.error.connect(self.handle_copy_errors)
